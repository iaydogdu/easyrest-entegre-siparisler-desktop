private setupWindowEvents(): void {

  // Ready to show
  this.mainWindow.once('ready-to-show', () => {
    console.log('üé¨ Pencere g√∂sterilmeye hazƒ±r');
    this.mainWindow?.show();
    this.mainWindow?.focus();
    
    // Development mode'da DevTools a√ß

      console.log('üîß DevTools a√ßƒ±ldƒ± (development mode)');

  this.mainWindow.on('closed', () => {
    console.log('‚ùå Ana pencere kapatƒ±ldƒ±');

  });

  // Minimize to tray
  this.mainWindow.on('minimize', (event) => {
    if (this.tray) {
      console.log('üì± Pencere tray\'e minimize edildi');

  });

  this.mainWindow.on('maximize', () => {
    console.log('üî≥ Pencere maximize edildi');
  });

  this.mainWindow.on('unmaximize', () => {
    console.log('üî≤ Pencere restore edildi');

    console.log(`üß≠ Navigation: ${url}`);
  });

    if (isDev) {
      console.log(`üñ•Ô∏è Renderer [${level}]: ${message}`);

}

private setupTray(): void {
  try {
    console.log('üì± System tray kuruluyor...');
    
    // Tray icon path
    const trayIconPath = this.getTrayIconPath();
    
    // Tray olu≈ütur

    this.tray.setToolTip('EasyRest Entegre Sipari≈üler - Desktop');

private setupTrayEvents(): void {
  if (!this.tray) return;

  // Double click to show
  this.tray.on('double-click', () => {
    console.log('üì± Tray double-click: Pencere g√∂steriliyor');
    this.showWindow();
  });

  // Right click for context menu (Windows/Linux)
  this.tray.on('right-click', () => {
    if (process.platform !== 'darwin') {
      this.tray?.popUpContextMenu();
    }
  });

  // Click event (macOS)
  this.tray.on('click', () => {
    if (process.platform === 'darwin') {
      this.showWindow();
    }
  });
}

private showWindow(): void {
  if (this.mainWindow) {
    if (this.mainWindow.isMinimized()) {
      this.mainWindow.restore();
    }
    
    this.mainWindow.show();
    this.mainWindow.focus();
    
    // macOS'ta Dock'ta g√∂ster
    if (process.platform === 'darwin') {
      app.dock?.show();
    }
  }
}

private updateTrayForNewOrder(orderData: any): void {
  if (!this.tray) return;

  console.log('üì± Tray yeni sipari≈ü bildirimi:', orderData.orderId);

  // Tray icon'u deƒüi≈ütir (alert state)
  const alertIconPath = path.join(__dirname, '../src/assets/icons/tray-icon-alert.png');
  this.tray.setImage(nativeImage.createFromPath(alertIconPath));

  // Tooltip g√ºncelle
  this.tray.setToolTip(`EasyRest - Yeni Sipari≈ü: ${orderData.orderId}`);

  // Flash animasyonu
  let flashCount = 0;
  const maxFlashes = 6;
  
  const flashInterval = setInterval(() => {
    const iconPath = flashCount % 2 === 0 
      ? alertIconPath
      : this.getTrayIconPath();
    
    this.tray?.setImage(nativeImage.createFromPath(iconPath));
    flashCount++;
    
    if (flashCount >= maxFlashes) {
      clearInterval(flashInterval);
      // Normal icon'a geri d√∂n
      this.tray?.setImage(nativeImage.createFromPath(this.getTrayIconPath()));
      this.tray?.setToolTip('EasyRest Entegre Sipari≈üler');
    }
  }, 500);

  // Balloon notification (Windows)
  if (process.platform === 'win32') {
    this.tray.displayBalloon({
      title: 'Yeni Sipari≈ü!',
      content: `${orderData.platform} - ${orderData.orderId}\nTutar: ${orderData.amount} ‚Ç∫`,
      icon: nativeImage.createFromPath(alertIconPath)
    });
  }
}

private resetTrayIcon(): void {
  if (!this.tray) return;
  
  const normalIconPath = this.getTrayIconPath();
  this.tray.setImage(nativeImage.createFromPath(normalIconPath));
  this.tray.setToolTip('EasyRest Entegre Sipari≈üler');
}

  globalShortcut.unregisterAll();

}

private setupAutoUpdaterEvents(): void {
  autoUpdater.on('checking-for-update', () => {
    console.log('üîç G√ºncellemeler kontrol ediliyor...');
    this.updateTrayTooltip('G√ºncellemeler kontrol ediliyor...');
  });

  autoUpdater.on('update-available', (info) => {
    console.log('üì• G√ºncelleme mevcut:', {
      version: info.version,
      releaseDate: info.releaseDate,
      size: info.files?.[0]?.size

  });

  autoUpdater.on('update-not-available', (info) => {
    console.log('‚úÖ G√ºncelleme mevcut deƒüil:', info.version);
  });

  autoUpdater.on('error', (err) => {
    console.error('‚ùå G√ºncelleme hatasƒ±:', err);
    this.showUpdateErrorDialog(err);
  });

  autoUpdater.on('download-progress', (progressObj) => {
    const percent = Math.round(progressObj.percent);
    console.log(`üì• ƒ∞ndirme ilerlemesi: ${percent}% (${progressObj.bytesPerSecond} B/s)`);
    
    this.updateTrayTooltip(`G√ºncelleme indiriliyor: ${percent}%`);
    
    // Progress notification
    if (percent % 25 === 0) { // Her %25'te bildir
      this.showDownloadProgress(percent);
    }
  });

  autoUpdater.on('update-downloaded', (info) => {
    console.log('‚úÖ G√ºncelleme indirildi:', info.version);
    this.showUpdateReadyDialog(info);
  });
}

    ];
